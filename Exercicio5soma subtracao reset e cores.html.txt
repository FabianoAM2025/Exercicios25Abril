  <h1>Aprendendo a somar e subtrair </h1>
  <p>Clique aqui para somar ou subtrair:</p>
  <button onclick="somar()">Clique aqui para somar:</button>  
  <button onclick="subtrair()">Clique aqui para subtrair:</button>
  <button onclick="reset()">Clique aqui para resetar tudo:</button>
  <div id="DIVsoma">Placar = 0</div>
  <div id="DIVsubtracao">Placar = 0</div>

  <script>
    let placar = 0;

    function somar() {
      placar += 1;
      document.getElementById("DIVsoma").innerText = "Placar = " + placar;
      document.getElementById("DIVsoma").style.color = "blue"; // Corrigido para 'color'
    }

    function subtrair() {
      placar -= 1;
      document.getElementById("DIVsubtracao").innerText = "Placar = " + placar;
      document.getElementById("DIVsubtracao").style.color = "red"; // Corrigido para 'color'
    }
    
    function reset() {
      placar = 0;
      document.getElementById("DIVsoma").innerText = "Placar = " + placar;
      document.getElementById("DIVsubtracao").innerText = "Placar = " + placar;
      document.getElementById("DIVsoma").style.color = "#800080"; // Cor resetada
      document.getElementById("DIVsubtracao").style.color = "#800080"; // Cor resetada
    }
  </script>
